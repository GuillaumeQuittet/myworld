Boost Ubuntu (on low ram computer) :
-------------------------------------
- Edit : gksu gedit /etc/sysctl.conf
- Add : vm.swappiness=X
X = 5 for less 1 Go of RAM
X = 10 for 2 - 4 Go of RAM
X = 20 for SSD and 4 - 32Go of RAM

Disable guest Session :
------------------------
sudo nano /usr/share/lightdm/lightdm.conf.d/50-ubuntu.conf

Add this line at the end : allow-guest=false

# Touchpad configuration
# /etc/X11/xorg.conf.d/50-synaptics.conf
# Synclient Options:
# -------------------
Section "InputClass"
    Identifier "touchpad"
    Driver "synaptics"
    MatchIsTouchpad "on"
        Option "TapButton1" "1"
        Option "TapButton2" "3"
        Option "TapButton3" "2"
        Option "ClickFinger1" "1"
        Option "ClickFinger2" "1"
        Option "ClickFinger3" "1"
        Option "VertScrollDelta" "-114"
        Option "HorizScrollDelta" "-114"
        Option "CircularScrolling" "on"
        Option "CircScrollTrigger" "2"
        Option "ClickPad" "true"
EndSection

# Keyboard configuration
# Live configuration : setxkbmap -layout us -variant altgr-intl -option nodeadkeys
# nodeadkeys option is just to disable dead keys but this option isn't mandatory.
# --------------------------------------------------------------------------------
# Persistant configuration
# -------------------------
# /etc/X11/xorg.conf.d/00-keyboard.conf
# ---------------------------------------
Section "InputClass"
    Identifier "system-keyboard"
    MatchIsKeyboard "on"
    Option "XkbLayout" "us"
    Option "XkbVariant" "altgr-intl"
    Option "XkbOptions" "compose:ralt, caps:escape, terminate:ctrl_alt_bksp"
EndSection
